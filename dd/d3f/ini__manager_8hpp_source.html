45constexpr auto trim(std::string_view str) noexcept -> std::string_view
46{
47 const size_t first = str.find_first_not_of(" \t\r\n");
48 if (std::string_view::npos == first)
49 {
50 return {};
51 }
52 const size_t last = str.find_last_not_of(" \t\r\n");
53 return str.substr(first, (last - first + 1));

252 template <typename T>
253 auto get_value(section section, key key) const noexcept -> std::optional<T>
254 {
255 if (auto value_str = (*this)[section.value][key.value]; value_str.has_value())
256 {
257 if constexpr (std::is_same_v<T, std::string>)
258 {
259 return value_str;
260 }
261 else if constexpr (std::is_same_v<T, bool>)
262 {
263 // Use a local copy for modification
264 std::string lower_value = *value_str;
265 std::ranges::transform(lower_value, lower_value.begin(), ::tolower);
266 // Trim potential whitespace around boolean value before comparison
267 lower_value = std::string{trim(lower_value)};
268
269 if (lower_value == "true" || lower_value == "1")
270 {
271 return true;
272 }
273 if (lower_value == "false" || lower_value == "0")
274 {
275 return false;
276 }
277 return std::nullopt;
278 }
279 else if constexpr (StreamExtractable<T>)
280 {
281 std::istringstream iss(*value_str);
282 T value;
283 // Check for successful extraction AND that the entire string was consumed
284 if ((iss >> value) && iss.eof())
285 {
286 return value;
287 }
288 }
289 }
290 return std::nullopt;
291 }

571 auto parse(std::istream &istream) -> std::expected<void, std::error_code>
572 {
573 std::string line;
574 std::optional<std::string> current_section;
575
576 while (std::getline(istream, line))
577 {
578 const std::string_view line_view = trim(line);
579
580 // Skip empty lines and comments
581 if (line_view.empty() || line_view.starts_with(';') ||
582 line_view.starts_with('#'))
583 {
584 continue;
585 }
586
587 // Section header: [SectionName]
588 if (line_view.starts_with('[') && line_view.ends_with(']'))
589 {
590 if (line_view.length() < 3)
591 { // Handle empty section "[ ]" -> treat as unnamed or error?
592 // Current behavior: treats as empty string section name ""
593 current_section = "";
594 }
595 else
596 {
597 current_section =
598 std::string{trim(line_view.substr(1, line_view.length() - 2))};
599 }
600 // Ensure the section exists in the map (creates if new)
601 set_section(*current_section);
602 continue;
603 }
604
605 // Key-value pair: Key = Value
606 if (auto delimiter_pos = line_view.find('=');
607 delimiter_pos != std::string_view::npos)
608 {
609 if (current_section.has_value())
610 {
611 auto key = trim(line_view.substr(0, delimiter_pos));
612 auto value = trim(line_view.substr(delimiter_pos + 1));
613 // Check if key is empty
614 if (!key.empty())
615 {
616 (*m_data)[*current_section][std::string{key}] =
617 std::string{value};
618 }
619 }
620 }
621
622 // Check stream state *after* processing the line
623 if (istream.fail() && !istream.eof())
624 {
625 return std::unexpected(std::error_code(EIO, std::system_category()));
626 }
627 }
628
629 // Check final stream state after loop (e.g., if badbit is set without failbit)
630 if (istream.bad())
631 {
632 return std::unexpected(std::error_code(EIO, std::system_category()));
633 }
634
635 return {};
636 }
637
643 auto write(std::ostream &ostream) const -> std::expected<void, std::error_code>
644 {
645 for (const auto &[section, entries] : (*m_data))
646 {
647 ostream << "[" << section << "]\n";
648 for (const auto &[key, value] : entries)
649 {
650 ostream << key << " = " << value << "\n";
651 }
652 ostream << "\n";
653 }
654 // Check stream state after writing all data
655 if (ostream.fail())
656 {
657 return std::unexpected(std::error_code(errno, std::system_category()));
658 }
