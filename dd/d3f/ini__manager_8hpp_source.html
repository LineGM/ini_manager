43constexpr auto trim(std::string_view str) noexcept -> std::string_view
44{
45 const size_t first = str.find_first_not_of(" \t\r\n");
46 if (std::string_view::npos == first)
47 {
48 return {};
49 }
50 const size_t last = str.find_last_not_of(" \t\r\n");
51 return str.substr(first, (last - first + 1));
52}

259 template <typename T>
260 auto get_value(section section, key key) const noexcept -> std::optional<T>
261 {
262 if (auto value_str = (*this)[section.value][key.value]; value_str.has_value())
263 {
264 if constexpr (std::is_same_v<T, std::string>)
265 {
266 return value_str;
267 }
268 else if constexpr (std::is_same_v<T, bool>)
269 {
270 std::istringstream iss(*value_str);
271 std::string lower_value;
272 iss >> lower_value;
273 std::ranges::transform(lower_value, lower_value.begin(), ::tolower);
274 if (lower_value == "true" || lower_value == "1")
275 {
276 return true;
277 }
278 if (lower_value == "false" || lower_value == "0")
279 {
280 return false;
281 }
282 return std::nullopt;
283 }
284 else if constexpr (StreamExtractable<T>)
285 {
286 std::istringstream iss(*value_str);
287 T value;
288 if (iss >> value && iss.eof())
289 {
290 return value;
291 }
292 }
293 }
294 return std::nullopt;
295 }

536 auto parse(std::istream &istream) -> std::expected<void, std::error_code>
537 {
538 std::string line;
539 std::optional<std::string> current_section;
540 while (std::getline(istream, line))
541 {
542 const std::string_view line_view = trim(line);
543
544 if (line_view.empty() || line_view.starts_with(';') ||
545 line_view.starts_with('#'))
546 {
547 continue;
548 }
549
550 if (line_view.starts_with('[') && line_view.ends_with(']'))
551 {
552 current_section =
553 std::string{trim(line_view.substr(1, line_view.length() - 2))};
554 continue;
555 }
556
557 if (auto delimiter_pos = line_view.find('=');
558 delimiter_pos != std::string_view::npos)
559 {
560 if (current_section.has_value())
561 {
562 auto key = trim(line_view.substr(0, delimiter_pos));
563 auto value = trim(line_view.substr(delimiter_pos + 1));
564 (*m_data)[*current_section][std::string{key}] = std::string{value};
565 }
566 }
567 if (istream.fail() && !istream.eof())
568 {
569 return std::unexpected(std::error_code(errno, std::system_category()));
570 }
571 }
572 return {};
573 }

580 auto write(std::ostream &ostream) const -> std::expected<void, std::error_code>
581 {
582 for (const auto &[section, entries] : (*m_data))
583 {
584 ostream << "[" << section << "]\n";
585 for (const auto &[key, value] : entries)
586 {
587 ostream << key << " = " << value << "\n";
588 }
589 ostream << "\n";
590 }
591 if (ostream.fail())

594 }
