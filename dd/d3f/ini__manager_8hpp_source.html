22constexpr auto trim(std::string_view str) noexcept -> std::string_view
23{
24 const size_t first = str.find_first_not_of(" \t\r\n");
25 if (std::string_view::npos == first)
26 {
27 return {};
28 }
29 const size_t last = str.find_last_not_of(" \t\r\n");
30 return str.substr(first, (last - first + 1));

152 auto get_value(section section, key key) const noexcept -> std::optional<T>
153 {
154 if (auto value_str = (*this)[section.value][key.value]; value_str.has_value())
155 {
156 if constexpr (std::is_same_v<T, std::string>)
157 {
158 return value_str;
159 }
160 else if constexpr (std::is_same_v<T, bool>)
161 {
162 std::istringstream iss(*value_str);
163 std::string lower_value;
164 iss >> lower_value;
165 std::ranges::transform(lower_value, lower_value.begin(), ::tolower);
166 if (lower_value == "true" || lower_value == "1")
167 {
168 return true;

171 {

325 auto parse(std::istream &istream) -> std::expected<void, std::error_code>
326 {
327 std::string line;
328 std::optional<std::string> current_section;
329 while (std::getline(istream, line))
330 {
331 const std::string_view line_view = trim(line);
332
333 if (line_view.empty() || line_view.starts_with(';') ||
334 line_view.starts_with('#'))
335 {
336 continue;
337 }
338
339 if (line_view.starts_with('[') && line_view.ends_with(']'))
340 {
341 current_section =
342 std::string{trim(line_view.substr(1, line_view.length() - 2))};
343 continue;
344 }
345
346 if (auto delimiter_pos = line_view.find('=');
347 delimiter_pos != std::string_view::npos)
348 {
349 if (current_section.has_value())
350 {
351 auto key = trim(line_view.substr(0, delimiter_pos));
352 auto value = trim(line_view.substr(delimiter_pos + 1));
353 (*m_data)[*current_section][std::string{key}] = std::string{value};
354 }
355 }
356 if (istream.fail() && !istream.eof())
357 {
358 return std::unexpected(std::error_code(errno, std::system_category()));
359 }
360 }
361 return {};
362 }
